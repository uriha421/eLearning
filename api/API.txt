users

		request				email username password
		response			success(ture or false),message
	email auth			GET		/user/:auth_password(128bit)	200,400,500
		response			success, message
								if true response session(id,username,icon_url)
	login				POST	/user/:username			200,400,500
		request				email or username
		response			success, message
								if true response session(id,username,icon_url)
	logout				DELETE	/user/:username			204,400,401,403,500
		response			success, message
	get user info		GET	/user/:username				200,400,401,403,500
		response			success, message
								if true response ※something
	change user info	PUT /user/:username				200,400,401,403,500
		request				※something
		response			success, message
	change password 	PUT /user/:username/password
		request				password
		response			success, message
